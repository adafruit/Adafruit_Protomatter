25 /*
26 Common ground for architectures to support this library:
27 
28 - 32-bit device (e.g. ARM core, but potentially others in the future)
29 - One or more 32-bit GPIO PORTs with atomic bitmask SET and CLEAR registers.
30  A TOGGLE register, if present, may improve performance but is NOT required.
31 - Tolerate 8-bit or word-aligned 16-bit accesses within the 32-bit PORT
32  registers (e.g. writing just one of four bytes, rather than the whole
33  32 bits). The library does not use any unaligned accesses (i.e. the
34  "middle word" of a 32-bit register), even if a device tolerates such.
35 
36 "Pin" as used in this code is always a uint8_t value, but the semantics
37 of what it means may vary between Arduino and non-Arduino situations.
38 In Arduino, it's the pin index one would pass to functions such as
39 digitalWrite(), and doesn't necessarily correspond to physical hardware
40 pins or any other arrangement. Some may have names like 'A0' that really
41 just map to higher indices.
42 In non-Arduino settings (CircuitPython, other languages, etc.), how a
43 pin index relates to hardware is entirely implementation dependent, and
44 how to get from one to the other is what must be implemented in this file.
45 Quite often an environment will follow the Arduino pin designations
46 (since the numbers are on a board's silkscreen) and will have an internal
47 table mapping those indices to registers and bitmasks...but probably not
48 an identically-named and -structured table to the Arduino code, hence the
49 reason for many "else" situations in this code.
50 
51 Each architecture defines the following macros and/or functions (the _PM_
52 prefix on each is to reduce likelihood of naming collisions...especially
53 on ESP32, which has some similarly-named timer functions:
54 
55 GPIO-related macros/functions:
56 
57 _PM_portOutRegister(pin): Get address of PORT out register. Code calling
58  this can cast it to whatever type's needed.
59 _PM_portSetRegister(pin): Get address of PORT set-bits register.
60 _PM_portClearRegister(pin): Get address of PORT clear-bits register.
61 _PM_portToggleRegister(pin): Get address of PORT toggle-bits register.
62  Not all devices support this, in which case
63  it must be left undefined.
64 _PM_portBitMask(pin): Get bit mask within PORT register corresponding
65  to a pin number. When compiling for Arduino,
66  this just maps to digitalPinToBitMask(), other
67  environments will need an equivalent.
68 _PM_byteOffset(pin): Get index of byte (0 to 3) within 32-bit PORT
69  corresponding to a pin number.
70 _PM_wordOffset(pin): Get index of word (0 or 1) within 32-bit PORT
71  corresponding to a pin number.
72 _PM_pinOutput(pin): Set a pin to output mode. In Arduino this maps
73  to pinMode(pin, OUTPUT). Other environments
74  will need an equivalent.
75 _PM_pinInput(pin): Set a pin to input mode, no pullup. In Arduino
76  this maps to pinMode(pin, INPUT).
77 _PM_pinHigh(pin): Set an output pin to a high or 1 state. In
78  Arduino this maps to digitalWrite(pin, HIGH).
79 _PM_pinLow(pin): Set an output pin to a low or 0 state. In
80  Arduino this maps to digitalWrite(pin, LOW).
81 
82 Timer-related macros/functions:
83 
84 _PM_timerFreq: A numerical constant - the source clock rate
85  (in Hz) that's fed to the timer peripheral.
86 _PM_timerInit(void*): Initialize (but do not start) timer.
87 _PM_timerStart(void*,count): (Re)start timer for a given timer-tick interval.
88 _PM_timerStop(void*): Stop timer, return current timer counter value.
89 _PM_timerGetCount(void*): Get current timer counter value (whether timer
90  is running or stopped).
91 A timer interrupt service routine is also required, syntax for which varies
92 between architectures.
93 The void* argument passed to the timer functions is some indeterminate type
94 used to uniquely identify a timer peripheral within a given environment. For
95 example, in the Arduino wrapper for this library, compiling for SAMD chips,
96 it's just a pointer directly to a timer/counter peripheral base address. If
97 an implementation needs more data associated alongside a peripheral, this
98 could instead be a pointer to a struct, or an integer index.
99 
100 Other macros/functions:
101 
102 _PM_chunkSize: Matrix bitmap width (both in RAM and as issued
103  to the device) is rounded up (if necessary) to
104  a multiple of this value as a way of explicitly
105  unrolling the innermost data-stuffing loops.
106  So far all HUB75 displays I've encountered are
107  a multiple of 32 pixels wide, but in case
108  something new comes along, or if a larger
109  unroll actually decreases performance due to
110  cache size, this can be set to whatever works
111  best (any additional data is simply shifted
112  out the other end of the matrix). Default if
113  unspecified is 8 (e.g. four loop passes on a
114  32-pixel matrix, eight if 64-pixel). Only
115  certain chunkSizes are actually implemented,
116  see .cpp code (avoiding GCC-specific tricks
117  that would handle arbitrary chunk sizes).
118 _PM_delayMicroseconds(us): Function or macro to delay some number of
119  microseconds. For Arduino, this just maps to
120  delayMicroseconds(). Other environments will
121  need to provide their own or map to an
122  an equivalent function.
123 _PM_clockHoldHigh: Additional code (typically some number of NOPs)
124  needed to delay the clock fall after RGB data is
125  written to PORT. Only required on fast devices.
126  If left undefined, no delay happens.
127 _PM_clockHoldLow: Additional code (e.g. NOPs) needed to delay
128  clock rise after writing RGB data to PORT.
129  No delay if left undefined.
130 _PM_minMinPeriod: Mininum value for the "minPeriod" class member,
131  so bit-angle-modulation time always doubles with
132  each bitplane (else lower bits may be the same).
133 */

322 #endif

339 #define NUM_TIMERS (sizeof timer / sizeof timer[0])

341  Tc *tc = (Tc *)tptr; // Cast peripheral address passed in
342 
343  uint8_t timerNum = 0;
344  while ((timerNum < NUM_TIMERS) && (timer[timerNum].tc != tc)) {
345  timerNum++;
346  }
347  if (timerNum >= NUM_TIMERS)
348  return;
349 
350  // Feed timer/counter off GCLK1 (already set 48 MHz by Arduino core).
351  // Sure, SAMD51 can run timers up to F_CPU (e.g. 120 MHz or up to
352  // 200 MHz with overclocking), but on higher bitplanes (which have
353  // progressively longer timer periods) I could see this possibly
354  // exceeding a 16-bit timer, and would have to switch prescalers.
355  // We don't actually need atomic precision on the timer -- point is
356  // simply that the period doubles with each bitplane, and this can
357  // work fine at 48 MHz.
358  GCLK->PCHCTRL[timer[timerNum].GCLK_ID].bit.CHEN = 0; // Disable
359  while (GCLK->PCHCTRL[timer[timerNum].GCLK_ID].bit.CHEN)
360  ; // Wait for it
361  GCLK_PCHCTRL_Type pchctrl; // Read-modify-store
362  pchctrl.reg = GCLK->PCHCTRL[timer[timerNum].GCLK_ID].reg;
363  pchctrl.bit.GEN = GCLK_PCHCTRL_GEN_GCLK1_Val;
364  pchctrl.bit.CHEN = 1;
365  GCLK->PCHCTRL[timer[timerNum].GCLK_ID].reg = pchctrl.reg;
366  while (!GCLK->PCHCTRL[timer[timerNum].GCLK_ID].bit.CHEN)
367  ;
368 
369  // Disable timer before configuring it
370  tc->COUNT16.CTRLA.bit.ENABLE = 0;
371  while (tc->COUNT16.SYNCBUSY.bit.ENABLE)
372  ;
373 
374  // 16-bit counter mode, 1:1 prescale
375  tc->COUNT16.CTRLA.bit.MODE = TC_CTRLA_MODE_COUNT16;
376  tc->COUNT16.CTRLA.bit.PRESCALER = TC_CTRLA_PRESCALER_DIV1_Val;
377 
378  tc->COUNT16.WAVE.bit.WAVEGEN =
379  TC_WAVE_WAVEGEN_MFRQ_Val; // Match frequency generation mode (MFRQ)
380 
381  tc->COUNT16.CTRLBCLR.reg = TC_CTRLBCLR_DIR; // Count up
382  while (tc->COUNT16.SYNCBUSY.bit.CTRLB)
383  ;
384 
385  // Overflow interrupt
386  tc->COUNT16.INTENSET.reg = TC_INTENSET_OVF;
387 
388  NVIC_DisableIRQ(timer[timerNum].IRQn);
389  NVIC_ClearPendingIRQ(timer[timerNum].IRQn);
390  NVIC_SetPriority(timer[timerNum].IRQn, 0); // Top priority
391  NVIC_EnableIRQ(timer[timerNum].IRQn);
392 
393  // Timer is configured but NOT enabled by default
394 }

1221  const uint16_t *lowerSrc =
1222  source + width * core->numRowPairs; // " bottom half
1223  uint8_t *pinMask = (uint8_t *)core->rgbMask; // Pin bitmasks
1224  uint8_t *dest = (uint8_t *)core->screenData;

1229 #if defined(_PM_portToggleRegister)
1230 #if !defined(_PM_STRICT_32BIT_IO)

1237 #endif

1239  // No need to clear matrix buffer, loops below do a full overwrite
1240  // (except for any scanline pad, which was already initialized in the
1241  // begin() function and won't be touched here).
1242 
1243  // Determine matrix bytes per bitplane & row (row pair really):
1244 
1245  uint32_t bitplaneSize =
1246  _PM_chunkSize *
1247  ((width + (_PM_chunkSize - 1)) / _PM_chunkSize); // 1 plane of row pair
1248  uint8_t pad = bitplaneSize - width; // Start-of-plane pad
1249 
1250  // Skip initial scanline padding if present (HUB75 matrices shift data
1251  // in from right-to-left, so if we need scanline padding it occurs at
1252  // the start of a line, rather than the usual end). Destination pointer
1253  // passed in already handles double-buffer math, so we don't need to
1254  // handle that here, just the pad...
1255  dest += pad;
1256 
1257  uint32_t initialRedBit, initialGreenBit, initialBlueBit;
1258  if (core->numPlanes == 6) {
1259  // If numPlanes is 6, red and blue are expanded from 5 to 6 bits.
1260  // This involves duplicating the MSB of the 5-bit value to the LSB
1261  // of its corresponding 6-bit value...or in this case, bitmasks for
1262  // red and blue are initially assigned to canvas MSBs, while green
1263  // starts at LSB (because it's already 6-bit). Inner loop below then
1264  // wraps red & blue after the first bitplane.
1265  initialRedBit = 0b1000000000000000; // MSB red
1266  initialGreenBit = 0b0000000000100000; // LSB green
1267  initialBlueBit = 0b0000000000010000; // MSB blue
1268  } else {

1270  // three color components might be decimated by some number of bits.
1271  // The initial bitmasks are set to the components' numPlanesth bit
1272  // (e.g. for 5 planes, start at red & blue bit #0, green bit #1,
1273  // for 4 planes, everything starts at the next bit up, etc.).
1274  uint8_t shiftLeft = 5 - core->numPlanes;
1275  initialRedBit = 0b0000100000000000 << shiftLeft;
1276  initialGreenBit = 0b0000000001000000 << shiftLeft;
1277  initialBlueBit = 0b0000000000000001 << shiftLeft;
1278  }

1280  // This works sequentially-ish through the destination buffer,
1281  // reading from the canvas source pixels in repeated passes,
1282  // beginning from the least bit.
1283  for (uint8_t row = 0; row < core->numRowPairs; row++) {
1284  uint32_t redBit = initialRedBit;
1285  uint32_t greenBit = initialGreenBit;
1286  uint32_t blueBit = initialBlueBit;
1287  for (uint8_t plane = 0; plane < core->numPlanes; plane++) {
1288 #if defined(_PM_portToggleRegister)
1289  uint8_t prior = clockMask; // Set clock bit on 1st out
1290 #endif
1291  for (uint16_t x = 0; x < width; x++) {
1292  uint16_t upperRGB = upperSrc[x]; // Pixel in upper half
1293  uint16_t lowerRGB = lowerSrc[x]; // Pixel in lower half
1294  uint8_t result = 0;
1295  if (upperRGB & redBit)
1296  result |= pinMask[0];
1297  if (upperRGB & greenBit)
1298  result |= pinMask[1];
1299  if (upperRGB & blueBit)
1300  result |= pinMask[2];
1301  if (lowerRGB & redBit)
1302  result |= pinMask[3];
1303  if (lowerRGB & greenBit)
1304  result |= pinMask[4];
1305  if (lowerRGB & blueBit)
1306  result |= pinMask[5];
1307 #if defined(_PM_portToggleRegister)
1308  dest[x] = result ^ prior;
1309  prior = result | clockMask; // Set clock bit on next out
1310 #else
1311  dest[x] = result;
1312 #endif
1313  } // end x
1314  greenBit <<= 1;
1315  if (plane || (core->numPlanes < 6)) {
1316  // In most cases red & blue bit scoot 1 left...
1317  redBit <<= 1;
1318  blueBit <<= 1;
1319  } else {
1320  // Exception being after bit 0 with 6-plane display,
1321  // in which case they're reset to red & blue LSBs
1322  // (so 5-bit colors are expanded to 6 bits).
1323  redBit = 0b0000100000000000;
1324  blueBit = 0b0000000000000001;
1325  }
1326 #if defined(_PM_portToggleRegister)
1327  // If using bit-toggle register, erase the toggle bit on the
1328  // first element of each bitplane & row pair. The matrix-driving
1329  // interrupt functions correspondingly set the clock low before
1330  // finishing. This is all done for legibility on oscilloscope --
1331  // so idle clock appears LOW -- but really the matrix samples on
1332  // a rising edge and we could leave it high, but at this stage
1333  // in development just want the scope "readable."
1334  dest[-pad] &= ~clockMask; // Negative index is legal & intentional
1335 #endif
1336  dest += bitplaneSize; // Advance one scanline in dest buffer
1337  } // end plane
1338  upperSrc += width; // Advance one scanline in source buffer
1339  lowerSrc += width;
1340  } // end row
1341 }
